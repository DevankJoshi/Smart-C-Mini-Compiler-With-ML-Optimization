Modify your compiler_driver.js to use the ML optimizer instead of the basic one:

// Find this in compile() function:
// Phase 7 — TAC Optimizer
try {
  const { optimizedIR, report } = optimizeTAC(ir);
  document.getElementById('out-7').innerHTML = renderOptimizerResult(ir, optimizedIR, report);
  activatePhase(7, false);
} catch(e) { ... }

// Replace with this:
// Phase 7 — ML-Guided TAC Optimizer
try {
  // Use ML-guided optimization
  const { optimizedIR, report } = optimizeTACWithML(ir, true);
  
  // Render ML prediction + optimization results
  const mlReport = renderMLPrediction(report);
  const optimReport = renderOptimizerResult(ir, optimizedIR, report);
  document.getElementById('out-7').innerHTML = mlReport + optimReport;
  
  // Generate optimized C code
  const optimizedC = tacToC(optimizedIR);
  
  // Show optimized code
  document.getElementById('out-7').innerHTML +=
    `<div style="margin-top:20px;background:rgba(16,185,129,0.08);
        border:1px solid rgba(16,185,129,0.3);border-radius:8px;padding:15px">
      <div style="font-size:11px;color:#10b981;font-weight:600;margin-bottom:10px">
        📝 Optimized C Code
      </div>
      <pre style="background:#1e293b;color:#e2e8f0;padding:12px;border-radius:4px;
          overflow-x:auto;font-size:12px">${escapeHtml(optimizedC)}</pre>
    </div>`;
  
  activatePhase(7, false);
} catch(e) {
  const out = document.getElementById('out-7');
  if (out) out.innerHTML = `<div class="alert-err">Optimizer error: ${esc(e.message)}</div>`;
  activatePhase(7, true);
}
⚠️ Note: Make sure you have the helper function escapeHtml(): 
function escapeHtml(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
}

let mlOptimizationEnabled = true;

function toggleMLOptimization() {
  mlOptimizationEnabled = !mlOptimizationEnabled;
  alert(`ML Optimization ${mlOptimizationEnabled ? 'enabled' : 'disabled'}`);
  compile(); // Re-run compilation
}

function downloadOptimizedCode() {
  const optimizedCode = document.querySelector('#out-7 pre')?.textContent;
  if (!optimizedCode) {
    alert('No optimized code available');
    return;
  }
  const blob = new Blob([optimizedCode], { type: 'text/plain' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'optimized.c';
  a.click();
}

function showOptimizationReport() {
  const src = document.getElementById('src-editor').value;
  const result = mlOptimizeCompile(src, true, true);
  const report = createOptimizationReport(src, result);
  alert(report);
}

Advanced: Custom Optimization Strategy

You can customize which optimizations to apply:

function customOptimize(ir, strategy) {
  let opt = cloneIR(ir);
  
  // Apply only selected passes
  if (strategy.constantFolding) {
    const res = passConstantFolding(opt);
    console.log('Constant folding:', res.hits.length, 'changes');
  }
  
  if (strategy.deadCode) {
    const res = passDeadCode(opt);
    console.log('Dead code removal:', res.hits.length, 'removals');
  }
  
  if (strategy.cse) {
    const res = passCSE(opt);
    console.log('CSE:', res.hits.length, 'eliminations');
  }
  
  return opt;
}

// Usage
const strategy = {
  constantFolding: true,
  deadCode: true,
  cse: true
};
const optimized = customOptimize(ir, strategy);

Troubleshooting

Issue: "extractFeatures is not defined"
Solution: Make sure feature_extractor.js is loaded before ml_optimizer.js
Issue: "Model weights are all zeros"
Solution: Pre-trained weights are included by default. If you trained custom weights, make sure you loaded them correctly.
Issue: "Optimized code looks wrong"
Solution: TAC→C conversion uses gotos. The code is semantically correct but may look unfamiliar. Check TAC view for details.
Issue: "Low optimization percentage"
Solution: Not all code patterns benefit from optimization. Try the example programs to see what works best.
